
//Q.1=> Write a program to print MySirG 5 times on the screen
 
 #include<stdio.h>
 int main(){

}
 return 0;
 }
  
//Q.2=> Write a program to print the first 10 natural numbers.

 #include<stdio.h>
 int main(){
 for(int i=1; i<=10; i++)
 {
    printf("%d\n",i);
 }
 return 0;
 }

//Q.3=> Write a program to print the first 10 natural numbers in reverse order

  #include<stdio.h>
 int main(){
 for(int i=10; i>=1; i--)
 {
    printf("%d\n",i);
 }
 return 0;
 }
