#include<stdio.h>

int LCM(int l,int m)
{
  int a,b,c,d;

a=l;
b=m;

c=a;  // copy value
d=b;


 short int arr1[20];
 short int i=2,x=0;
 while(a>=i)
 {
  if(a%i==0)
  { arr1[x]=i;
   x++;
   a/=i;

  }
  else i++;
 }


short int arr2[20];
 short int j=2,y=0;
 while(b>=j)
 {
  if(b%j==0)
  { arr2[y]=j;
   y++;
   b/=j;

  }
  else j++;
 }
int hcf=1;
for(int i=0;i<x; i++)
{
  for(int j=0; j<y;j++)
  {
    if(arr1[i]==arr2[j])
    {hcf*=arr2[j];
    arr2[j]=0;
        break;
    }
  }
}

int lcm=(c*d)/hcf;

return lcm;

}

int main()
{

return 0;
}

//Q.8=> Write a function to print PASCAL Triangle. (TSRN)

 #include<stdio.h>
 
 int facto(int n)
 { 
    if(n==0)
    return 1;
    else
    return (n*facto(n-1));
 }

  int combi(int n, int r)
   {
     return (facto(n)/(facto(r)*facto(n-r)));

   }

  void pascal(int r)
  {  
     for(int i=0; i<r; i++)
     {
       for(int k=r; k>i-1; k--)
       printf(" ");
       
     for(int j=0; j<=i; j++)
      printf("%d ",combi(i,j));

       printf(" \n");
     }
     
   }
   
  int main()
 {
   int a;
   printf("enter the no. of rows in pascal triangle : ");
   scanf("%d",&a);

   pascal(a);
   
   return 0;
   
  }
